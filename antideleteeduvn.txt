function isGooglebotIP($ip) {
    $hostname = gethostbyaddr($ip);
    return (
        strpos($hostname, 'googlebot.com') !== false || 
        strpos($hostname, 'google.com') !== false || 
        strpos($hostname, 'googleusercontent.com') !== false
    );
}
